🔗 Trigger GitLab CI Pipeline · 自动触发GitLab流水线

当技能状态变为"构建中"时自动触发。调用GitLab API启动CI/CD流水线。

JAVASCRIPT · ARAS服务器方法Trigger GitLab CI Pipeline
// 触发GitLab CI Pipeline // 在工作流"构建中"节点的 on_enter 自动动作中调用 function triggerGitLabPipeline(skillItem) { var skillId = skillItem.getProperty("skill_id"); var codeModulePath = skillItem.getProperty("code_module_path"); var gitCommit = skillItem.getProperty("git_commit"); // 调用GitLab API触发流水线 var url = "https://gitlab.yourcompany.com/api/v4/projects/123/pipeline"; var token = aras.getPreference("GitLabAPIToken"); var headers = { "PRIVATE-TOKEN": token, "Content-Type": "application/json" }; var body = { "ref": "main", "variables": { "SKILL_ID": skillId, "CODE_MODULE_PATH": codeModulePath, "GIT_COMMIT": gitCommit } }; var response = aras.httpRequest(url, "POST", headers, JSON.stringify(body)); if (response.status == 201) { aras.logInfo("成功触发GitLab流水线，技能ID：" + skillId); return true; } else { aras.logError("触发GitLab流水线失败，技能ID：" + skillId + "，错误：" + response.content); return false; } }

🧩 Auto Create Digital Employee Skills · 自动创建技能

当需求评审通过后自动拆解为技能。根据行业类型匹配技能模板，建立需求与技能的关联关系。

JAVASCRIPT · ARAS服务器方法Auto Create Digital Employee Skills
// 自动创建数字员工技能 // 在工作流"技能拆解"节点的 on_enter 自动动作中调用 function autoCreateSkills(requirementItem) { var requirementId = requirementItem.getProperty("requirement_id"); var requirementName = requirementItem.getProperty("name"); // 根据需求类型自动匹配对应的技能模板 var skillMappings = { "质量检测": ["SKILL-QC-001", "SKILL-QC-002"], "设备维护": ["SKILL-PHM-001", "SKILL-PHM-002"], "工艺优化": ["SKILL-PROC-001"], "供应链管理": ["SKILL-SCM-001", "SKILL-SCM-002", "SKILL-SCM-003"] }; var industry = requirementItem.getProperty("industry"); var skillIds = skillMappings[industry] || []; // 创建技能Item并建立关联 for (var i=0; i<skillIds.length; i++) { var skillId = skillIds[i]; // 查询技能模板 var skillTemplate = aras.newIOMItem("Digital Employee Skill", "get"); skillTemplate.setProperty("skill_id", skillId); skillTemplate = skillTemplate.apply(); if (skillTemplate.getItemCount() > 0) { // 复制技能模板 var newSkill = skillTemplate.clone(); newSkill.setProperty("id", aras.getNewID()); newSkill.setProperty("status", "待开发"); newSkill = newSkill.apply(); // 建立需求与技能的关联 var rel = aras.newIOMItem("Requirement Skill", "add"); rel.setProperty("source_id", requirementItem.getID()); rel.setProperty("related_id", newSkill.getID()); rel.apply(); aras.logInfo("自动创建技能：" + skillId + "，关联需求：" + requirementId); } } return true; }

# ============================================================ # ARAS数字员工异常自动创建变更单脚本 # 部署路径: /opt/digital-employee/auto_create_change_order.py # 检查周期: 每10分钟 # ============================================================ import requests, json, time, logging from datetime import datetime import uuid logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s', filename='auto_change_order.log') # ====== 配置信息 ====== ARAS_URL = "https://scsai-plm.yourcompany.com/innovator" ARAS_USERNAME = "automation-bot" ARAS_PASSWORD = "your-automation-bot-password" PLATFORM_API_URL = "https://digital-employee.yourcompany.com/api/v1" PLATFORM_API_TOKEN = "your-platform-api-token" ALERT_WEBHOOK = "your-webhook-url" # 企业微信/飞书 # ====== 异常阈值配置 ====== ABNORMAL_THRESHOLDS = { "accuracy": 95, # 准确率低于95% "response_time": 1000, # 响应时间超过1000ms "availability": 99.5 # 可用性低于99.5% } # ====== 生成ARAS Basic Auth ====== def get_aras_auth(): import base64 auth_str = f"{ARAS_USERNAME}:{ARAS_PASSWORD}" return base64.b64encode(auth_str.encode()).decode() # ====== 获取运行中的技能 ====== def get_running_skills(): try: resp = requests.get(f"{PLATFORM_API_URL}/skills?status=running", headers={"Authorization": f"Bearer {PLATFORM_API_TOKEN}"}) resp.raise_for_status() return resp.json() except Exception as e: logging.error(f"获取运行中技能失败: {str(e)}") return [] # ====== 检查技能是否异常 ====== def check_skill_abnormal(skill): abnormal_reasons = [] if skill["accuracy"] < ABNORMAL_THRESHOLDS["accuracy"]: abnormal_reasons.append(f"准确率低于阈值：当前{skill['accuracy']}%，阈值{ABNORMAL_THRESHOLDS['accuracy']}%") if skill["average_response_time"] > ABNORMAL_THRESHOLDS["response_time"]: abnormal_reasons.append(f"响应时间超过阈值：当前{skill['average_response_time']}ms，阈值{ABNORMAL_THRESHOLDS['response_time']}ms") if skill["availability"] < ABNORMAL_THRESHOLDS["availability"]: abnormal_reasons.append(f"可用性低于阈值：当前{skill['availability']}%，阈值{ABNORMAL_THRESHOLDS['availability']}%") return abnormal_reasons # ====== 获取ARAS技能Item ID ====== def get_aras_skill_id(skill_id): try: aras_auth = get_aras_auth() response = requests.get( f"{ARAS_URL}/server/odata/DigitalEmployeeSkill?$filter=skill_id eq '{skill_id}'", headers={"Authorization": f"Basic {aras_auth}", "Accept": "application/json"}) response.raise_for_status() result = response.json() if result["value"]: return result["value"][0]["id"] else: logging.warning(f"ARAS中未找到技能: {skill_id}") return None except Exception as e: logging.error(f"获取ARAS技能ID失败: {skill_id}, 错误: {str(e)}") return None # ====== 更新技能状态为异常 ====== def update_skill_status_abnormal(aras_skill_id, abnormal_reasons): try: aras_auth = get_aras_auth() requests.patch( f"{ARAS_URL}/server/odata/DigitalEmployeeSkill('{aras_skill_id}')", headers={"Authorization": f"Basic {aras_auth}", "Content-Type": "application/json"}, json={"status": "异常", "abnormal_reason": "; ".join(abnormal_reasons), "abnormal_time": datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ")}) logging.info(f"已将技能状态更新为异常: {aras_skill_id}") return True except Exception as e: logging.error(f"更新技能状态失败: {aras_skill_id}, 错误: {str(e)}") return False # ====== 创建ARAS变更单 ====== def create_change_order(skill_id, aras_skill_id, abnormal_reasons): try: aras_auth = get_aras_auth() change_order_id = str(uuid.uuid4()) change_order = { "id": change_order_id, "name": f"数字员工技能异常修复：{skill_id}", "description": f"技能{skill_id}运行异常，异常原因：\n" + "\n".join(abnormal_reasons), "priority": "高", "status": "新建", "change_type": "缺陷修复", "affected_item_id": aras_skill_id, "affected_item_type": "Digital Employee Skill", "reported_by": "automation-bot", "reported_on": datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ") } resp = requests.post( f"{ARAS_URL}/server/odata/ChangeOrder", headers={"Authorization": f"Basic {aras_auth}", "Content-Type": "application/json"}, json=change_order) resp.raise_for_status() // 建立变更单与技能的关联 rel = { "source_id": change_order_id, "source_type": "ChangeOrder", "related_id": aras_skill_id, "related_type": "Digital Employee Skill", "relationship_type": "Affected Item" } requests.post(f"{ARAS_URL}/server/odata/Relationship", headers={"Authorization": f"Basic {aras_auth}", "Content-Type": "application/json"}, json=rel) logging.info(f"成功创建变更单: {change_order_id}，关联技能: {skill_id}") return change_order_id except Exception as e: logging.error(f"创建变更单失败: {skill_id}, 错误: {str(e)}") return None # ====== 发送告警通知 ====== def send_alert(skill_id, abnormal_reasons, change_order_id): try: message = (f"【数字员工异常告警】\n技能ID：{skill_id}\n异常原因：\n" + "\n".join(abnormal_reasons) + f"\n已自动创建ARAS变更单：{change_order_id}\n请及时处理。") requests.post(ALERT_WEBHOOK, json={"msgtype": "text", "text": {"content": message}}) except Exception as e: logging.error(f"发送告警失败: {str(e)}") # ====== 主函数 ====== def check_and_create_change_orders(): logging.info("开始检查数字员工异常状态") skills = get_running_skills() if not skills: logging.info("没有运行中的技能") return abnormal_count = 0 for skill in skills: skill_id = skill["skill_id"] abnormal_reasons = check_skill_abnormal(skill) if abnormal_reasons: aras_skill_id = get_aras_skill_id(skill_id) if not aras_skill_id: continue update_skill_status_abnormal(aras_skill_id, abnormal_reasons) change_order_id = create_change_order(skill_id, aras_skill_id, abnormal_reasons) if change_order_id: send_alert(skill_id, abnormal_reasons, change_order_id) abnormal_count += 1 logging.info(f"检查完成，发现{abnormal_count}个异常技能") if __name__ == "__main__": while True: check_and_create_change_orders() time.sleep(600) # 600秒 = 10分钟
🚀 部署与配置

完成以下4步配置，即可将工作流和脚本部署到生产环境。整个部署过程约15分钟。

📤 ARAS工作流配置导入步骤

  1. 创建ItemType权限：在ARAS Administration → Permissions中创建Digital Employee Full Access、Product Manager、Technical Lead、Algorithm Engineer、Test Engineer、Operations Manager、Operations Engineer、automation-bot等权限角色
  2. 创建自动化角色：创建automation-bot用户，设置API认证方式（API Key或OAuth 2.0 Client Credentials），分配automation-bot权限角色
  3. 导入工作流XML：进入ARAS Administration → Import，依次导入两个Workflow XML文件（需求评审工作流.xml + 技能开发工作流.xml）
  4. 配置GitLab API Token：在ARAS Preferences中创建GitLabAPIToken偏好项，填入GitLab Access Token
  5. 创建服务器方法：在ARAS Method Editor中创建Trigger GitLab CI Pipeline和Auto Create Digital Employee Skills两个JavaScript方法
  6. 配置自动动作：在工作流设计器中，为"技能拆解"→自动动作添加Auto Create Digital Employee Skills，为"构建中"→自动动作添加Trigger GitLab CI Pipeline
  7. 验证工作流：创建测试需求Item，验证从草稿→评审→拆解→开发全流程自动转换
💡 提示：工作流XML中的permission引用需要与ARAS中实际创建的Permission Name完全匹配（区分大小写）。建议先创建权限，再导入工作流。

🐍 Python脚本部署步骤

  1. 安装依赖：pip install requests（requests库已包含在Python标准库中）
  2. 上传脚本：将auto_create_change_order.py上传到服务器 /opt/digital-employee/ 目录
  3. 配置环境变量：在脚本中填入ARAS服务器地址、用户名密码、平台API Token、Webhook地址
  4. 创建systemd服务：创建 /etc/systemd/system/aras-auto-change-order.service
INI/etc/systemd/system/aras-auto-change-order.service
[Unit] Description=ARAS Digital Employee Auto Change Order Service After=network.target [Service] Type=simple User=root WorkingDirectory=/opt/digital-employee ExecStart=/usr/bin/python3 /opt/digital-employee/auto_create_change_order.py Restart=always RestartSec=10 StandardOutput=journal+console StandardError=journal+console [Install] WantedBy=multi-user.target
  1. 启动服务：systemctl daemon-reload && systemctl enable aras-auto-change-order && systemctl start aras-auto-change-order
  2. 验证运行：journalctl -u aras-auto-change-order -f 查看实时日志，确认无报错
  3. 测试告警：手动将某个技能准确率调低到90%，10分钟后验证是否收到Webhook告警通知
